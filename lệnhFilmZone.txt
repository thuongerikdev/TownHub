# 1) TẮT path conversion của MSYS
export MSYS_NO_PATHCONV=1
# (hoặc) export MSYS2_ARG_CONV_EXCL='*'

# 2) Tạo cert tự ký (OK trong Git Bash sau khi tắt path conv)
openssl req -x509 -newkey rsa:2048 -nodes \
  -keyout key.pem -out cert.pem -days 365 \
  -subj "/CN=filmzone.fly.dev"

# 3) Đóng gói .pfx
openssl pkcs12 -export -out server.pfx -inkey key.pem -in cert.pem \
  -passout pass:changeit

# 4) Base64 (Git Bash hỗ trợ -w0)
base64 -w0 server.pfx > server.pfx.b64

# 5) Đẩy vào Fly secrets
flyctl secrets set \
  KESTREL_PFX_BASE64="$(cat server.pfx.b64)" \
  KESTREL_PFX_PASSWORD="changeit" \
  -a filmzone-api
